# Pascal&#039;s triangle
[Challenge Link](https://tsch.js.org/30958)
## Challenge

Given a number N, construct the Pascal's triangle with N rows.
[Wikipedia](https://en.wikipedia.org/wiki/Pascal%27s_triangle)

## Solution

```ts
type FromLength<
  N extends number,
  Acc extends 1[] = [],
> = N extends Acc["length"] ? Acc : FromLength<N, [...Acc, 1]>;

type Sum<A extends number, B extends number> = [
  FromLength<A>,
  FromLength<B>,
] extends [infer ArrA extends 1[], infer ArrB extends 1[]]
  ? [...ArrA, ...ArrB]["length"]
  : never;

type SumArr<
  A extends number[],
  B extends number[],
  Acc extends number[] = [],
> = [A, B] extends [
  [infer HeadA extends number, ...infer TailA extends number[]],
  [infer HeadB extends number, ...infer TailB extends number[]],
]
  ? SumArr<TailA, TailB, [...Acc, Sum<HeadA, HeadB>]>
  : Acc;

type GetLast<T extends unknown[]> = [never, ...T][T["length"]];

type GetNextRow<T extends number[]> = SumArr<[0, ...T], [...T, 0]>;

type PascalArr<
  N extends 1[],
  Acc extends number[][] = [[1]],
  LastOfAcc extends number[] = GetLast<Acc>,
> = N["length"] extends 1
  ? Acc
  : N extends [1, ...infer Tail extends 1[]]
  ? GetNextRow<LastOfAcc> extends infer NewRow extends number[]
    ? PascalArr<Tail, [...Acc, NewRow]>
    : never
  : never;

type Pascal<N extends number> = FromLength<N> extends infer ArrN extends 1[]
  ? PascalArr<ArrN>
  : never;
```
